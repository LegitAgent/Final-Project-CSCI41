SELECT QUERIES of LISTED activities 

Gets Activity of user 
SELECT central_bookings_activity.name, 
account_user.name, 
central_bookings_reservation.date, 
central_bookings_reservation.start_time, 
central_bookings_reservation.end_time, 
central_bookings_location.name, 
central_bookings_organizer.name, 
central_bookings_activitybooking.attended
FROM central_bookings_activity,
central_bookings_activitybooking, 
account_user, central_bookings_reservation, 
central_bookings_location, 
central_bookings_organizer
WHERE 
central_bookings_activity.id = central_bookings_activitybooking.activity_id 
AND account_user.id = central_bookings_activitybooking.participant_id 
AND central_bookings_reservation.activity_id = central_bookings_activity.id
AND central_bookings_location.id = central_bookings_reservation.location_id
AND central_bookings_organizer.id = central_bookings_activity.organizer_id;
